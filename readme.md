## Тема 3. Модульное программирование. ПЗ DB API 2.0


# Create venv:
    python3 -m venv venv

# Activate venv:
## In Windows:
    venv\Scripts\activate
     
## In macOS or Linux:
    source venv/bin/activate

# install library
    pip install -r requirements.txt

# start 
    ./venv/bin/python "main.py"
